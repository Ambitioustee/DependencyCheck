21  public class OssIndexAnalyzerTest extends BaseTest {
22  
23      @Test
24      public void should_enrich_be_included_in_mutex_to_prevent_NPE()
25              throws AnalysisException, MalformedPackageURLException {
26  
27          // Given
28          OssIndexAnalyzer analyzer = new SproutOssIndexAnalyzer();
29  
30  
31          Identifier identifier = new PurlIdentifier("maven", "test", "test", "1.0",
32                  Confidence.HIGHEST);
33  
34          Dependency dependency = new Dependency();
35          dependency.addSoftwareIdentifier(identifier);
36          Settings settings = getSettings();
37          Engine engine = new Engine(settings);
38          engine.setDependencies(Collections.singletonList(dependency));
39  
40          analyzer.initialize(settings);
41  
42          String expectedOutput = "https://ossindex.sonatype.org/component/pkg:maven/test/test@1.0";
43  
44          // When
45          analyzer.analyzeDependency(dependency, engine);
46  
47          // Then
48          assertTrue(identifier.getUrl().startsWith(expectedOutput));
49      }
50  
51      /*
52       * This action is inspired by the sprout method technique displayed in
53       * "Michael Feathers - Working Effectively with Legacy code".
54       *
55       * We want to trigger a race condition between a call to
56       * OssIndexAnalyzer.closeAnalyzer() and OssIndexAnalyzer.enrich().
57       *
58       * The last method access data from the "reports" field while
59       * closeAnalyzer() erase the reference. If enrich() is not included in
60       * the "FETCH_MUTIX" synchronized statement, we can trigger a
61       * NullPointeException in a multithreaded environment, which can happen
62       * due to the usage of java.util.concurrent.Future.
63       *
64       * We want to make sure enrich() will be able to set the url of an
65       * identifier and enrich it.
66       */
67      static final class SproutOssIndexAnalyzer extends OssIndexAnalyzer {
68          @Override
69          void enrich(Dependency dependency) {
70              ExecutorService executor = Executors.newSingleThreadExecutor();
71              executor.execute(() -> {
72                  try {
73                      this.closeAnalyzer();
74                  } catch (Exception e) {
75                      throw new RuntimeException(e);
76                  }
77              });
78              super.enrich(dependency);
79          }
80      }
81  }
